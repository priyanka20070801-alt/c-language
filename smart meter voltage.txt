#include <stdio.h>
int main() {
    int N;
    scanf("%d", &N);
    int voltage;
    int minvoltage;
    int lowcount = 0;
    for (int i = 0; i < N; i++) {
        scanf("%d", &voltage); 
        if (i == 0) {
            minvoltage = voltage;
        } else if (voltage < minvoltage) {
            minvoltage = voltage;
        } if (voltage < 210) {
            lowcount++;
        }
    } printf("minvoltage: %d\n", minvoltage);
    printf("low voltage events: %d\n", lowcount);

    return 0;
}
